// دالة التحقق من كود الشراء عبر Envato API
function verify_envato_purchase($purchase_code) {
    // ضع هنا التوكن الخاص بك كمطور في انفاتو (Personal Token)
    $bearer_token = "YOUR_ENVATO_PERSONAL_TOKEN_HERE"; 
    
    $ch = curl_init();
    curl_setopt($ch, CURLOPT_URL, "https://api.envato.com/v3/market/author/sale?code=" . trim($purchase_code));
    curl_setopt($ch, CURLOPT_HTTPHEADER, [
        "Authorization: Bearer " . $bearer_token,
        "User-Agent: Purchase Code Verifier"
    ]);
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
    curl_setopt($ch, CURLOPT_TIMEOUT, 20);
    
    $response = curl_exec($ch);
    $http_status = curl_getinfo($ch, CURLINFO_HTTP_CODE);
    curl_close($ch);

    if ($http_status === 200) {
        $data = json_decode($response, true);
        // التأكد من أن الكود ينتمي لنفس منتجك (يمكنك مطابقة الـ item_id)
        if (isset($data['item']['id']) && $data['item']['id'] == "YOUR_CODECANYON_ITEM_ID") {
            return true;
        }
        // كود تجريبي مؤقت للتطوير (احذفه عند الرفع النهائي لـ CodeCanyon)
        if ($purchase_code === "TEST-1234-5678") return true; 
    }
    return false;
}

        if (empty($purchase_code)) {
            $error = "مفتاح الترخيص مطلوب لإتمام التثبيت.";
        } elseif (verify_envato_purchase($purchase_code)) {
            $_SESSION['purchase_code'] = $purchase_code;

        }
    }
    
    if ($step === 3) {
        $db_host = $_POST['db_host'] ?? 'localhost';
        $db_user = $_POST['db_user'] ?? '';
        $db_pass = $_POST['db_pass'] ?? '';
        $db_name = $_POST['db_name'] ?? '';
        
        try {
            // اختبار الاتصال بقاعدة البيانات
            $test_pdo = new PDO("mysql:host=$db_host;dbname=$db_name;charset=utf8mb4", $db_user, $db_pass);
            $test_pdo->setAttribute(PDO::ATTR_ERRMODE, PDO::ERRMODE_EXCEPTION);
            
            // قراءة ملف الـ SQL المطور وزرعه (تأكد من وجود ملف database.sql في مجلد install)
            $sql = file_get_contents(__DIR__ . '/database.sql');
            $test_pdo->exec($sql);
            
            // تحديث كود الشراء وحالة التثبيت داخل جدول الإعدادات
            $stmt = $test_pdo->prepare("UPDATE site_settings SET setting_value = ? WHERE setting_key = 'purchase_code'");
            $stmt->execute([$_SESSION['purchase_code'] ?? '']);
            
            $stmt2 = $test_pdo->prepare("UPDATE site_settings SET setting_value = '1' WHERE setting_key = 'is_installed'");
            $stmt2->execute();
            
            // إنشاء ملف الـ config.php الرئيسي ديناميكياً
            $config_content = "<?php\n"
                            . "define('DB_HOST', '$db_host');\n"
                            . "define('DB_USER', '$db_user');\n"
                            . "define('DB_PASS', '$db_pass');\n"
                            . "define('DB_NAME', '$db_name');\n"
                            . "define('IS_INSTALLED', true);\n";
            
            file_put_contents(__DIR__ . '/../includes/db_connect.php', $config_content); // أو مسار ملف الاتصال لديك
            
            $success = "تم تثبيت قاعدة البيانات بنجاح وإنشاء ملف الإعدادات!";
            header("Location: index.php?step=4");
            exit;
            
        } catch (PDOException $e) {
            $error = "فشل الاتصال بقاعدة البيانات: " . $e->getMessage();
        }
    }
}
